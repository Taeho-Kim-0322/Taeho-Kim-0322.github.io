## 한글 폰트 설정하기  
    ```
    #Win
    plt.rc('font', family="Malgun Gothic")
    plt.rc('axes', unicode_minus=False)
    #Mac
    # plt.rc('font', family="AppleGothic")

    # 폰트가 선명하게 보이도록 설정
    from IPython.display import set_matplotlib_formats
    set_matplotlib_formats('retina')

    # 한글폰트와 마이너스 폰트 설정 확인
    plt.title('한글폰트 설정')
    plt.plot([-4, -6, 1,2,3])
    ```
